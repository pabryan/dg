$$
\gamma(s) = \frac{1}{\sqrt{2}} (\cos s, \sin s, s)
$$

<p class="fragment">
$$
T = \frac{1}{\sqrt{2}} (-\sin s, \cos s, 1)
$$
</p>

<p class="fragment">
$$
N = \frac{\partial_s T}{\lvert \partial_s T\rvert} = (-\cos s, -\sin s, 0)
$$
</p>

<p class="fragment">
$$
B = T \times N = \frac{1}{\sqrt{2}} (\sin s, -\cos s, 1)
$$
</p>

<p class="fragment">
$$
\kappa \equiv \tau \equiv \frac{1}{\sqrt{2}}
$$
</p>
